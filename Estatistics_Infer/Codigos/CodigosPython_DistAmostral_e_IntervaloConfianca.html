Teorema do limite central¶

\begin{equation}\text{Se X} \sim (\mu, \sigma^2) \text{ tem-se que, } \bar{X} \sim N(\mu,\frac{ \sigma^2}{n}), {n \to \infty }\\\end{equation}

# Parmetros
num_samples = 1000
sample_sizes = [1, 30, 1000]

# Crie uma funo para gerar amostras de diferentes distribuies
def generate_samples(dist_name, sample_size):
    if dist_name == "binomial":
        return np.random.binomial(n = 5, p=0.2, size=sample_size)
    elif dist_name == "gamma":
        return np.random.gamma(shape=0.5,scale=2,size = sample_size)
    elif dist_name == "uniform":
        return np.random.uniform(1, 10, size=sample_size)

# Crie os grficos
for dist_name in ["binomial", "gamma", "uniform"]:
    plt.figure(figsize=(12, 10))
    for i, sample_size in enumerate(sample_sizes):
        plt.subplot(2, 3, i + 1)
        samples = [generate_samples(dist_name, sample_size) for _ in range(num_samples)]
        sample_means = np.mean(samples, axis=1)
        
        # Trace o histograma das mdias amostrais
        sns.histplot(sample_means, label='Sample Means')    
        
        plt.title(f'Sample Size: {sample_size}')
        plt.xlabel('Sample Mean')
        plt.ylabel('Frequency')
        plt.legend()

    plt.suptitle(f'{dist_name.capitalize()} Distribution - Sample Means with Normal Distribution', fontsize=16)
    plt.tight_layout()
    plt.subplots_adjust(top=0.9)

    plt.show()

def IC_normal_sigma_conhecido(amostra, desvio_padrao_pop, nivel_confianca): 
    media_amostral = np.mean(amostra)
    n = len(amostra)
    nivel_confianca = nivel_confianca

    print("O tamanho da amostra  de: ", n, " pacotes de caf.")
    print("A mdia amostral : ",media_amostral)

    IC_media = st.norm.interval(nivel_confianca, loc=media_amostral, scale=desvio_padrao_pop/np.sqrt(n))

    print("O intervalo de ", nivel_confianca*100 ,
          "% confiana para a quantidade mdia de gramas nos pacotes de caf  de: ", [round(num,2) for num in IC_media])
    
    return(IC_media)

def IC_normal_sigma_desconhecido(amostra, nivel_confianca, print_infos= True): 
    media_amostral = np.mean(amostra)
    desvio_amostral = np.std(amostra, ddof = 1) #desvio padro AMOSTRAL 
    n = len(amostra)
    
    if (print_infos): 
        print("O tamanho da amostra  de: ", n, " crianas.")
        print("A mdia amostral : ", round(media_amostral,1) ,'anos')
        print("O desvio padro amostral : ", round(desvio_amostral,1)  ,'anos')

    IC_media_t = st.t.interval(nivel_confianca, df = n-1,  loc=media_amostral, scale=desvio_amostral/np.sqrt(n))

    print("O intervalo de ", nivel_confianca*100 ,
          "% confiana para a idade mdia ao falar  de: ", [round(num,2) for num in IC_media_t])
    
    return(IC_media_t)

def IC_proporcao_conservador(n, sucesso, nivel_confianca):
    p = sucesso/n
        
    # Use the conservative (exact) method
    standard_error = 1 / (2 * math.sqrt(n))

    # Calculate the critical value for the desired confidence level
    alpha = 1 - nivel_confianca
    critical_value = st.norm.ppf(1 - alpha / 2)

    # Calculate the margin of error
    margin_of_error = critical_value * standard_error

    # Calculate the lower and upper bounds of the confidence interval
    lower_bound = p - margin_of_error
    upper_bound = p + margin_of_error
    
    return(lower_bound,upper_bound)

#IC de 95% de confiana
