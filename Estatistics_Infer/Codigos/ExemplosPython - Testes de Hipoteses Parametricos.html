from scipy.stats import norm
import numpy as np

def two_sample_ztest(sample1, sample2, sigma1_squared, sigma2_squared, alpha=0.05, alternative="two-sided"):
    """
    Realiza um teste de hipteses para as mdias de duas populaes independentes com varincias conhecidas.

    Argumentos:
    - sample1, sample2: As amostras das duas populaes.
    - sigma1_squared, sigma2_squared: As varincias conhecidas das populaes.
    - alpha: Nvel de significncia (padro: 0.05).
    - alternative: Especifica a hiptese alternativa. Pode ser "two-sided" (padro), "less" (menor) ou "greater" (maior).

    Retorna:
    - estatistica_teste: A estatstica de teste.
    - p_valor: O valor p do teste de hipteses.
    - rejeitar_nula: Um booleano indicando se a hiptese nula  rejeitada.
    - intervalo_confianca: O intervalo de confiana para a diferena das mdias.
    - valor_critico: O valor crtico para o teste de hiptese com base na hiptese alternativa.
    """
    # Tamanho das amostras
    n1, n2 = len(sample1), len(sample2)

    # Mdias das amostras
    mean1, mean2 = np.mean(sample1).round(0), np.mean(sample2).round(0)

    # Diferena das mdias
    mean_difference = mean1 - mean2

    # Desvio padro combinado
    combined_std = np.sqrt((sigma1_squared / n1) + (sigma2_squared / n2)).round(6)

    # Estatstica de teste
    z_statistic = mean_difference / combined_std

    # Teste de hipteses
    if alternative == "two-sided":
        p_value = 2 * (1 - norm.cdf(np.abs(z_statistic)))
        valor_critico = norm.ppf(1 - alpha / 2)
    elif alternative == "less":
        p_value = norm.cdf(z_statistic)
        valor_critico = norm.ppf(alpha)
    elif alternative == "greater":
        p_value = 1 - norm.cdf(z_statistic)
        valor_critico = norm.ppf(1 - alpha)
    else:
        raise ValueError("O argumento 'alternative' deve ser 'two-sided', 'less' ou 'greater'.")

    # Prints solicitados
    print("O tamanho da amostra 1  de:", n1)
    print("O tamanho da amostra 2  de:", n2)
    print("A varincia da populao  1 :", sigma1_squared)
    print("A varincia da populao 2 :", sigma2_squared)
    print("Valor Crtico:", valor_critico.round(3))

    # Testar a hiptese nula
    print("Estatstica de teste:", z_statistic.round(3))
    print("P-valor:", p_value.round(3))
    if(p_value <  alpha):
      print("Rejeitamos a Hiptese Nula")
    else:
      print("No rejeitamos a Hiptese Nula")

    return z_statistic, p_value, valor_critico

Teste de normalidade dos dados¶

A funo shapiro testa a hiptese nula $H_0$ de que a amostra proveniente de uma distribuio normal.

$$\begin{cases} H_0 = \text{Dados vem de uma normal}\\H_1 = \text{Dados no vem de uma normal}\end{cases} $$

def teste_normalidade_e_conclusao(dados, Pop, nivel_significancia=0.05):
    # Realizar o teste de normalidade de Shapiro-Wilk
    stat_test, p_valor = shapiro(dados)
    
    # Exibir os resultados
    print(f'\n{Pop} - Teste de Shapiro-Wilk: Estatstica de Teste = {round(stat_test, 2)}, P-valor = {round(p_valor, 4)}')
    
    # Concluso
    if p_valor < nivel_significancia:
        print(f"Rejeita-se a hiptese nula. Os dados da {Pop} no seguem uma distribuio normal.")
    else:
        print(f"No h evidncias para rejeitar a hiptese nula. Os dados da {Pop} podem ser considerados aproximadamente normais.")

Escolha entre Bartlett e Levene:

A escolha entre Bartlett e Levene depende das caractersticas dos seus dados. Se voc tem razo para acreditar que suas amostras vm de distribuies normais, o teste de Bartlett pode ser a escolha. Se voc est menos certo sobre a normalidade dos dados, ou se tem razes para acreditar que as amostras no seguem distribuies normais, o teste de Levene pode ser uma escolha mais robusta. Em geral, uma boa prtica realizar ambos os testes e considerar os resultados em conjunto.

from itertools import combinations

def testes_t_pareados_bonferroni(df, coluna_grupo, coluna_valor, alfa = 0.05):
    """
    Realiza testes t-pareados independentes entre todos os pares de grupos em um DataFrame.
    Ajusta os valores de p utilizando a correo de Bonferroni.

    Parmetros:
    - df: DataFrame contendo os dados
    - coluna_grupo: Nome da coluna contendo rtulos de grupo
    - coluna_valor: Nome da coluna contendo valores numricos
    """

    # Cria uma lista nica de grupos
    grupos = df[coluna_grupo].unique()
    testes_t = []
    
    # Gera todas as combinaes nicas de pares de grupos
    combinacoes = combinations(grupos, 2)

    # Itera pela lista de grupos
    for grupo1, grupo2 in combinacoes:
        tipo_1 = df[df[coluna_grupo] == grupo1]
        tipo_2 = df[df[coluna_grupo] == grupo2]

        t, p = st.ttest_ind(tipo_1[coluna_valor], tipo_2[coluna_valor])

        # Adiciona cada resultado de teste t a uma lista de pares t, p
        testes_t.append([grupo1 + ' - ' + grupo2, t.round(4), p.round(4)])

        # Imprime os resultados
        print(f'Resultados teste t: {grupo1}/{grupo2} - t: {round(t, 2)}, p: {round(p, 4)}')
        
    # O novo limiar para significncia estatstica = 0.05 / comprimento da lista de testes t
    alfa_ajustado = alfa/ len(testes_t)
    print(f'\n P-valor do teste t abaixo do alfa ajustado: {round(alfa_ajustado,4)}')

    # Imprime apenas os resultados qaue so significativos com base no novo limiar
    for par in testes_t:
        if par[2] <= alfa_ajustado:  # O valor p est no segundo ndice, por isso  2
            print(par)

Teste de esfericidade dos dados usando o pacote pingouin¶

Esfericidade a condio em que as varincias das diferenas entre todas as combinaes de grupos (nveis) relacionados so iguais. A violao da esfericidade ocorre quando as variaes das diferenas entre todas as combinaes de grupos relacionados no so iguais.

$$\begin{cases} H_0 = \text{Existe esfericidade}\\H_1 = \text{No existe esfericidade}\end{cases} $$

In [44]:
# https://pingouin-stats.org/build/html/generated/pingouin.sphericity.html

import pingouin as pg
spher, _, chisq, dof, pval =  pg.sphericity(data=df, dv='Nota', subject='Aluno', within=['Professor'], 
                                            alpha = nivel_significancia)

# Exibir os resultados de forma mais informativa
print(f"Teste de Mauchly para Esfericidade:")
print(f"Esfericidade Atingida: {spher}")
print(f"Estatstica Qui-Quadrado: {round(chisq, 3)}")
print(f"Graus de Liberdade: {dof}")
print(f"Valor-p: {round(pval, 3)}")

# Decidir sobre a esfericidade com base no valor-p
if pval < 0.05:
    print("Rejeitar H0: Esfericidade No Atingida")
else:
    print("Aceitar H0: Esfericidade Atingida")

O que fazer se a esfericidade for violada¶

Em cenrios onde o valor p inferior a 0,05 e rejeitamos a hiptese nula do teste de esfericidade de Mauchly, normalmente aplicamos uma correo aos graus de liberdade usados para calcular o ndice F.

Existem trs correes que podemos aplicar:

  • Huynh-Feldt (menos conservador)
  • Greenhouse-Geisser
  • Lower-bound (mais conservador)

Cada uma dessas correes tende a aumentar os valores de p na tabela de resultados da ANOVA de medidas repetidas para explicar o fato de que a suposio de esfericidade violada.

Ajuste da Anova¶

$$\begin{cases} H_0: \mu_1 = \mu_2 = \mu_3 = \mu_4\text{ (todas as mdias so iguais)}\\H_1: \text{Pelo menos uma diferente das demais} \end{cases} $$

$$\begin{cases} H_0 = \text{A diferena da mdia dos 4 grupos} = 0 \\H_1 = \text{A diferena da mdia dos 4 grupos} \neq 0 \end{cases} $$
